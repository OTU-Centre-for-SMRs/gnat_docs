VoidTransportMaterial

buildconstruction:Undocumented Class

The VoidTransportMaterial has not been documented. The content listed below should be used as a starting point for documenting the class, which includes the typical automatic documentation associated with a MooseObject; however, what is contained is ultimately determined by what is necessary to make the documentation clear for users.

A void material for regions with no neutron interactions. Provides the neutron group velocity () for transport problems and forces maximum void stabilization for the SAAF scheme. The properties must be listed in decreasing order by energy.

[Mesh]
  # Declare a mesh to solve SN neutron transport on. Here, we use the same mesh that Cardinal uses to
  # generate MGXS.
  [NeutronicsDomain]
    type = FileMeshGenerator
    file = mesh_in.e
  []
[]

# This is a custom input block in Gnat that sets up the SN transport problem, where we essentially tell
# Gnat how we want it to solve our problem. It's similar to OpenMCCellAverageProblem in Cardinal.
[TransportSystems]
  [Neutron]
    # We specify the number of energy groups here. This should match the number of energy groups
    # you specified when generating MGXS in Cardinal
    num_groups = 3

    # We specify the maximum degree of angular flux moments for the transport system to add. This
    # should be less than or equal to the number of scattering cross section Legendre moments
    # generated by Cardinal.
    max_anisotropy = 0

    # You should always set these two. The first is our numerical scheme (the Self-Adjoint Angular Flux formalism)
    # and the second is the type of particle we're simulating (neutron).
    scheme = saaf_cfem
    particle_type = neutron

    # The name of the output scalar flux variables (/ higher order angular flux moments).
    flux_moment_names = 'Flux_Moment'

    # Fission is still technically "experimental" in Gnat, so you have to specifically enable it
    # with this flag.
    debug_disable_fission = false

    # This flag says we're solving the eigenvalue (k_eff) form of the neutron transport equation.
    # Setting constant_ic = '1.0' fixes a bug where the problem fails to converge.
    eigen = true
    constant_ic = '1.0'

    # The spatial discretization variable type (family) and the spatial discretization order. Unless
    # you really know what you're doing, always use LAGRANGE and FIRST.
    family = LAGRANGE
    order = FIRST

    # The number of angles to use for the SN approximation.
    n_azimuthal = 3
    n_polar = 3

    # You'd set vacuum / reflective boundary conditions with these flags. They must correspond to sidesets
    # in the mesh.
    vacuum_boundaries = '10000'
    reflective_boundaries = '10001'
  []
[]

# In this problem, Cardinal doesn't generate MGXS in the air regions as the statistics are terrible.
# We approximate air as void with a TransportMaterial that sets all cross sections to zero.
[TransportMaterials/Void]
  type = VoidTransportMaterial
  block = 'air src src_trimmer_tri'
[]

# This is the block that imports cross sections from Cardinal.
[CardinalMGXS]
  # Since we're doing this "online", we select that we want to pull cross sections from a sub-application.
  # Another option is to load them from a mesh with an exodus restart.
  xs_source = 'subapp'
  # The subapp we want to pull cross sections from.
  from_multi_app = 'cardinal_xs'

  # The transport system we want to apply cross sections to
  transport_system = 'Neutron'
  # The order of scattering anisotropy. This should match the order declared in the MGXS block in Cardinal.
  scatter_anisotropy = 0
  # A flag to say that we want to add fission heating.
  add_fission_heating = true

  # The blocks to apply cross sections to. This is all non-air blocks in this example.
  block = 'fuel cladding graphite_core graphite_reflector stainless_steel
           fuel_trimmer_tri graphite_core_trimmer_tri graphite_reflector_trimmer_tri
           stainless_steel_trimmer_tri'
[]
